47  template <typename MatrixT, typename IndexSeqT>
48  typename IndexSeqT::iterator
49  partition_idx(const MatrixT& mat, IndexSeqT& filter) const {
50  return std::partition(
51  filter.begin(),
52  filter.end(),
53  [this, &mat](const size_t row){
54  return mat(row, this->m_split_col) <= this->m_split_val;
55  });
56  }

58  // total squared error for left and right side of split_val
59  template <typename MatrixT, typename VectorT>
60  double calc_total_err(
61  const MatrixT& xs,
62  const VectorT& ys,
63  const SizeVec& filter) const {
64  if (filter.empty()) {
65  return doubleNaN;
66  }
67 
68  const auto avg = partitioned_mean(xs, ys, filter);
69  const float yhat_l = avg.first;
70  const float yhat_r = avg.second;
71 
72  if (std::isnan(yhat_l) || std::isnan(yhat_r)) {
73  return doubleNaN;
74  }
75 
76  double err = 0;
77  for (const auto & row : filter) {
78  const auto x_j = xs(row, m_split_col);
79  const auto y_j = ys[row];
80  const auto yhat_j = x_j <= m_split_val ? yhat_l : yhat_r;
81  err += pow((y_j - yhat_j), 2.0);
82  }
83  return err;
84  }

86  private:

88  size_t m_split_col = 0;

90  template <typename MatrixT, typename VectorT>
91  std::pair<float, float>
92  partitioned_mean(
93  const MatrixT& xs,
94  const VectorT& ys,
95  const SizeVec& rows) const {
96  SizeVec part(rows);
97  const auto pivot = partition_idx(xs, part);
98 
99  auto first_p = part.begin();
100  auto last_p = part.end();
101  // std::optional would be good here, but not at C++ 17 yet
102  // for this install
103  if (first_p == last_p || pivot == first_p || pivot == last_p) {
104  return std::make_pair(floatNaN, floatNaN);
105  }
106 
107  const float yhat_l = mean(ys, first_p, pivot);
108  if (std::isnan(yhat_l)) {
109  return std::make_pair(floatNaN, floatNaN);
110  }
111 
112  const float yhat_r = mean(ys, pivot, last_p);
113  if (std::isnan(yhat_r)) {
114  return std::make_pair(floatNaN, floatNaN);
115  }
116  return std::make_pair(yhat_l, yhat_r);
117  }
118  };

120  template <typename MatrixT, typename VectorT>
121  SplitPoint
122  best_split(const MatrixT& xs, const VectorT& ys, const SizeVec& filter) {
123  SplitPoint best;
124 
125  // TODO min size guard
126  if (filter.empty()) {
127  return best;
128  }
129  double best_err = doubleNaN;
130 
131  const auto ncols = xs.ncol();
132  for (size_t split_col = 0; split_col != ncols; ++split_col) {
133  auto uniques = unique_x(xs, split_col, filter);
134 
135  if (uniques.size() < 2) {
136  continue;
137  }
138 
139  for (const auto & split_val : uniques) {
140  SplitPoint split(split_val, split_col);
141 
142  // total squared error for left and right side of split_val
143  const auto err = split.calc_total_err(xs, ys, filter);
144 
145  // TODO randomly allow the same error as best to 'win'
146  if (!std::isnan(err) && (std::isnan(best_err) || err < best_err)) {
147  best = std::move(split);
148  best_err = err;
149  }
150  }
151  }
152  return best;
153  }
