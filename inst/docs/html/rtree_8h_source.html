42  template <typename MatrixT, typename VectorT>
43  RTree(const MatrixT& xs, const VectorT& ys, const SizeVec& filter) {
44  if (filter.empty()) {
45  throw std::invalid_argument("Must have at least one entry");
46  }
47 
48  m_yhat = mean(ys, filter.begin(), filter.end());
49  if (std::isnan(m_yhat)) {
50  throw std::logic_error("Prediction is NaN");
51  }
52 
53  SplitPoint split;
54  if (variance(ys, filter.begin(), filter.end()) > 1e-6) {
55  split = best_split(xs, ys, filter);
56 
57  if (split.is_valid()) {
58  m_split = split;
59  m_is_leaf = false;
60 
61  SizeVec part(filter);
62  const auto pivot = m_split.partition_idx(xs, part);
63  SizeVec lsplit(part.begin(), pivot);
64  SizeVec rsplit(pivot, part.end());
65 
66  // stuck on C++ 11 b/c the Debian Rcpp build forces it?
67  // TODO I would like to fix that
68  m_left.reset(new RTree(xs, ys, lsplit));
69  m_right.reset(new RTree(xs, ys, rsplit));
70  }
71  }
72 
73  if (!m_is_leaf) {
74  if (!m_left) {
75  throw std::logic_error("Cannot have a null left child node");
76  }
77  if (!m_right) {
78  throw std::logic_error("Cannot have a null right child node");
79  }
80  }
81  }

102  template <typename MatrixT>
103  void
104  predict(const MatrixT& xs, SizeVec& filter, FloatVec& yhat) const {
105  if (m_is_leaf) {
106  for (const auto & row : filter) {
107  yhat[row] = m_yhat;
108  }
109  } else {
110  const auto pivot = m_split.partition_idx(xs, filter);
111  SizeVec lsplit(filter.begin(), pivot);
112  SizeVec rsplit(pivot, filter.end());
113 
114  m_left->predict(xs, lsplit, yhat);
115  m_right->predict(xs, rsplit, yhat);
116  }
117  }
